                        fullWidth
                        value={filter.max_year}
                        slotProps={{htmlInput: {min: filter.min_year, max: baseYearRange[1], step: 1}}}
                        onChange={e => filterAction({key: 'max_year', value: e.target.value})}
                      />
                    </Stack>
                  )}
                  <Tooltip
                    title={
                      'How to handle values from multiple years; display all, or select' +

                    placement="left"

                      <InputLabel id="time_agg_select">Time Handling</InputLabel>
                      <Select
                        labelId="time_agg_select"
                        label="Time Handling"
                        size="small"
                        value={view.time_agg}
                        onChange={e => {
                          viewAction({key: 'time_agg', value: e.target.value as 'last'})
                        }}
                      >
                        <MenuItem value="all" disabled={!view.as_plot}>
                          All Year
                        </MenuItem>
                        <MenuItem value="first">Earliest Year</MenuItem>
                        <MenuItem value="specified">Specified Year</MenuItem>
                        <MenuItem value="last">Latest Year</MenuItem>
                        <MenuItem value="mean">Average Across Years</MenuItem>
                      </Select>
                    </FormControl>
                  </Tooltip>
                  <Button onClick={() => filterAction({key: 'reset'})}>Reset Filters</Button>
                </Stack>
              </Box>
            </Stack>
          </CardContent>
          <CardActions>
            <Button
              fullWidth
              color="error"
              onClick={() => {
                window.history.replaceState(void 0, '', window.location.origin + window.location.pathname)
                window.location.reload()
              }}
            >
              Hard Reset
            </Button>
          </CardActions>
        </Card>
      </Drawer>
    </>
  )
